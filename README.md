#### 安装教程
```xml
    <repositories>
        <repository>
            <id>maven-repo</id>
            <url>https://gitee.com/ijr/ings-vip-mybaitsplus/tree/master/repo/ings/vip/mybaitsplus</url>
        </repository>
    </repositories>

    <dependency>
        <groupId>ings.vip</groupId>
        <artifactId>mybaitsplus</artifactId>
        <version>0.0.1-SNAPSHOT</version>
    </dependency>
```
